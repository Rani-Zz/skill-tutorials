class Solution
{
private:
    string s;
    map<char,int> mp;
public:
    /**
    *应该不是最优的
    */
  //Insert one char from stringstream
    void Insert(char ch)
    {
         s+=ch;
         mp[ch]++;
    }
  //return the first appearence once char in current stringstream
    char FirstAppearingOnce()
    {
        char res = '#';
      
        for(int i = 0;i<s.size();i++)
        {
            if(mp[s[i]]==1)
            {
                res = s[i];
                break;
            }  
        }
        return res;
    
    }

};
